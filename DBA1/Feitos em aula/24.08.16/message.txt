SELECT NOME, DNUMERO, SALARIO
	FROM FUNCIONARIO
    ORDER BY DNUMERO ASC, SALARIO DESC, NOME;
    
SELECT DNUMERO
	FROM FUNCIONARIO
    WHERE SEXO = 'M';
    
SELECT DISTINCT DNUMERO
	FROM FUNCIONARIO
    WHERE SEXO = 'M';
    
SELECT distinct SALARIO
	FROM FUNCIONARIO
    WHERE SEXO = 'F'
    ORDER BY SALARIO DESC;
    
SELECT distinct BAIRRO, CIDADE
	FROM FUNCIONARIO
    WHERE CIDADE = "São Paulo"
    ORDER BY BAIRRO;
    
SELECT NOME, SALARIO
	FROM FUNCIONARIO
    WHERE DNUMERO > 3;
    
SELECT NOME, DNOME
	FROM FUNCIONARIO, DEPARTAMENTO;
    
    
-- <> AND != IS THE SAME
SELECT CPF, NOME, DNUMERO
	FROM FUNCIONARIO
	WHERE DNUMERO <> 5; -- !=
    
SELECT NOME, SALARIO
	FROM FUNCIONARIO
	WHERE SALARIO <= 4000
    ORDER BY SALARIO DESC;
    
SELECT *
	FROM ATRIBUICAO
    WHERE HORAS > 2.0
    ORDER BY HORAS DESC;
    
SELECT NOME, DATANASC, CEP
	FROM FUNCIONARIO
	WHERE CEP LIKE '%12345%';
    
SELECT NOME, SALARIO, SEXO
	FROM FUNCIONARIO
    WHERE NOME LIKE 'J%';
    
SELECT NOME, RUA, NUMERO, BAIRRO, CIDADE, ESTADO, CEP, COMPL, CPF
	FROM FUNCIONARIO
    WHERE CPF LIKE '%3_';
    
SELECT NOME as 'Nome', SALARIO * 12 AS 'Salário Anual'
	FROM FUNCIONARIO
    WHERE SEXO = 'M';
    
SELECT 
	NOME AS 'Nome',
	SALARIO AS 'Salário',
	(SALARIO + 1000) AS 'Salário + 1000',
    (SALARIO * 1.1) AS 'Salário + 10%'
	FROM FUNCIONARIO
    ORDER BY SALARIO DESC;
    
SELECT
	NOME AS 'Nome',
    TIMESTAMPDIFF(YEAR, DATANASC, NOW()) AS 'Idade'
    FROM FUNCIONARIO
    ORDER BY IDADE DESC;
    
SELECT NOME, SALARIO, DNUMERO
	FROM FUNCIONARIO
    WHERE DNUMERO = 4
    AND SALARIO > 4000
    ORDER BY NOME;
    
SELECT NOME, SALARIO, DNUMERO
	FROM FUNCIONARIO
    WHERE DNUMERO = 4
    OR DNUMERO = 1;
    
-- SAME THING USE 'IN'
    
SELECT NOME, SALARIO, DNUMERO
	FROM FUNCIONARIO
    WHERE DNUMERO IN (1, 4);
    
SELECT NOME, SALARIO, DNUMERO
	FROM FUNCIONARIO
    WHERE DNUMERO NOT IN (1, 4);
    
SELECT NOME, CIDADE
	FROM FUNCIONARIO
    WHERE CIDADE IN ('São Paulo', 'Santo André')
    ORDER BY CIDADE;
    
SELECT NOME, SALARIO
	FROM FUNCIONARIO
    WHERE SALARIO >= 3000
    AND SALARIO <= 4500
    ORDER BY SALARIO DESC;
    
-- SAME THING USING 'BETWEEN'
    
SELECT NOME, SALARIO
	FROM FUNCIONARIO
    WHERE SALARIO BETWEEN 3000 AND 4500
    ORDER BY SALARIO DESC;
    
SELECT NOME, RUA, NUMERO, BAIRRO, CIDADE, ESTADO, CEP, COMPL
	FROM FUNCIONARIO
    WHERE COMPL IS NULL -- OR USE 'NOT NULL' TO GET NOT NULL VALUES  |  IN MYSQL 'COMPL = NULL' DON'T WORK
    ORDER BY NOME;
    
SELECT *
	FROM ATRIBUICAO
